115

118 // index for r cut-off
119 const auto icut =
120 std::size_t(std::distance(

123 stride;
124
125 std::vector<double> tv, tr;
126 tv.resize(icut);

129#pragma omp parallel for

131 // nb: Use H -> H+V (instead of H-> H-V), so change sign!
132 tv[i] = -f(m_Z, r[i * stride], m_rN);
