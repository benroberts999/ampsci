210inline double cg_2(int two_j1, int two_m1, int two_j2, int two_m2, int two_J,
211 int two_M)
212// <j1 m1, j2 m2 | J M> = (-1)^(j1-j2+M) * std::sqrt(2J+1) * (j1 j2 J)
213// . (m1 m2 -M)
214// (Last term is 3j symbol)
215// Note: this function takes INTEGER values, that have already multiplied by 2!
216// Works for l and j (integer and half-integer)
217{
218 if (triangle(two_j1, two_j2, two_J) * sumsToZero(two_m1, two_m2, -two_M) == 0)
219 return 0;
220 int sign = -1;
221 if ((two_j1 - two_j2 + two_M) % 4 == 0)
222 sign = 1; // mod 4 (instead 2), since x2
223 return sign * std::sqrt(two_J + 1.) *
224 gsl_sf_coupling_3j(two_j1, two_j2, two_J, two_m1, two_m2, -two_M);
225}

229inline bool sixj_zeroQ(int a, int b, int c, int d, int e, int f) {
230 // nb: works for 2*integers ONLY
231 // check triangle consitions
232 // Note: there are some 6j symbols that pass this test, though are still zero
233 // GSL calculates these to have extremely small (but non-zero) value
234 if (triangle(a, b, c) == 0)
235 return true;
236 if (triangle(c, d, e) == 0)
237 return true;
238 if (triangle(b, d, f) == 0)
239 return true;
240 if (triangle(e, f, a) == 0)
241 return true;
242 if (!evenQ(a + b + c))
243 return true;
244 if (!evenQ(c + d + e))
245 return true;
246 if (!evenQ(b + d + f))
247 return true;
248 if (!evenQ(e + f + a))
249 return true;
250 return false;
251}

306inline double Ck_kk(int k, int ka, int kb)
307// Reduced (relativistic) angular ME:
308// <ka||C^k||kb> = (-1)^(ja+1/2) * srt([ja][jb]) * 3js(ja jb k, -1/2 1/2 0) * Pi
309// Note: takes in kappa! (not j!)
310{
311 if (parity(l_k(ka), l_k(kb), k) == 0) {
312 return 0.0;
313 }
314 auto two_ja = twoj_k(ka);
315 auto two_jb = twoj_k(kb);
316 // auto sign = ((two_ja + 1) / 2 % 2 == 0) ? 1 : -1;
317 auto sign = evenQ_2(two_ja + 1) ? 1 : -1;
318 auto f = std::sqrt((two_ja + 1) * (two_jb + 1));
319 auto g = special_threej_2(two_ja, two_jb, 2 * k);
320 return sign * f * g;
321}

349inline std::pair<int, int> kminmax_Ck(int ka, int kb) {
350 // j = |k|-0.5
351 // kmin = |ja-jb| = | |ka| - |kb| |
352 // kmax = ja+jb = |ka| + |kb| - 1
353 const auto aka = std::abs(ka);
354 const auto akb = std::abs(kb);
355
356 auto kmin = std::abs(aka - akb);
357 auto kmax = aka + akb - 1;
358 // account for parity
359 if (parity(l_k(ka), l_k(kb), kmin) == 0)
360 ++kmin;
361 if (parity(l_k(ka), l_k(kb), kmax) == 0)
362 --kmax;
363
364 return {kmin, kmax};
365}

379inline double S_kk(int ka, int kb) {
380 auto la = l_k(ka);
381 if (la != l_k(kb))
382 return 0.0;
383 auto tja = twoj_k(ka);
384 auto tjb = twoj_k(kb);
385 if (triangle(tja, 2, tjb) == 0)
386 return 0;
387 auto sign = (((tja + 2 * la + 3) / 2) % 2 == 0) ? 1 : -1;
388 auto f = std::sqrt((tja + 1) * (tjb + 1) * 1.5);
389 // auto sixj = gsl_sf_coupling_6j(tja, 2, tjb, 1, 2 * la, 1);
390 auto sixj_sign = ((ka + kb) % 2 == 0) ? 1.0 : -1.0;
391 auto sixj = 0.5 * sixj_sign *
392 std::sqrt(std::fabs(((ka - 1) * (ka - 1) - kb * kb) /
393 (3.0 * ka * (1.0 + 2.0 * ka))));
394 return sign * f * sixj;
395}
