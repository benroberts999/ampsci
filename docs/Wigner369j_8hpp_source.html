9/*!
10 @brief
11 Angular provides functions and classes for calculating and storing angular
12 factors (3,6,9-J symbols etc.)
13
14 @details
15 Provides functions to:
16 - Calculate wigner 3,6,9-J symbols + Clebsh-Gordon coefs etc..
17 - Also provied look-up tables, which are faster than calculating symbols
18 on-the-fly
19 - Wrapper functions to calculate wigner 3,6,9-J symbols.
20 - Uses GSL:
21 https://www.gnu.org/software/gsl/doc/html/specfunc.html#coupling-coefficients
22
23 The general equations are defined:
24 \f{align}{
25 \frac{1}{r_{12}} &= \sum_{kq} \frac{r_<^k}{r_>^{k+1}}(-1)^q
26 C^k_{-q}(\hat{n}_1)C^k_{q}(\hat{n}_2)\\
27 C^k_{q} &\equiv \sqrt{\frac{4\pi}{2k+1}} Y_{kq}(\hat{n}),
28 \f}
29 and
30 \f{align}{
31 C^k_{ab} &\equiv \langle{\kappa_a}||C^k||{\kappa_b}\rangle
32 \equiv (-1)^{j_a+1/2} \widetilde C^k_{ab} ,\\
33 &= (-1)^{j_a+1/2}\sqrt{[j_a][j_b]}
34 \begin{pmatrix}
35 {j_a} & {j_b} & {k} \\ {-1/2} & {1/2} &{0}
36 \end{pmatrix}
37 \pi(l_a+l_b+k).
38 \f}
39*/

42//==============================================================================
43//! returns l given kappa
44constexpr int l_k(int ka) { return (ka > 0) ? ka : -ka - 1; }
45//! returns 2j given kappa
46constexpr int twoj_k(int ka) { return (ka > 0) ? 2 * ka - 1 : -2 * ka - 1; }
47//! returns parity [(-1)^l] given kappa
48constexpr int parity_k(int ka) {
49 return (ka % 2 == 0) ? ((ka > 0) ? 1 : -1) : ((ka < 0) ? 1 : -1);
50}
51//! returns parity [(-1)^l] given l
52constexpr int parity_l(int l) { return (l % 2 == 0) ? 1 : -1; }
53//! "Complimentary" l := (2j-l) = l+/-1, for j=l+/-1/2
54constexpr int l_tilde_k(int ka) { return (ka > 0) ? ka - 1 : -ka; }
55//! returns kappa, given 2j and l
56constexpr int kappa_twojl(int twoj, int l) {
57 return ((2 * l - twoj) * (twoj + 1)) / 2;
58}
59//! returns kappa, given 2*j and parity (+/-1),
60constexpr int kappa_twojpi(const int twoj, const int pi) {
61 const auto l_minus = (twoj - 1) / 2;
62 const auto l_minus_pi = (l_minus % 2 == 0) ? 1 : -1;
63 const auto l = l_minus_pi == pi ? l_minus : l_minus + 1;
64 return kappa_twojl(twoj, l); // must be simpler way?
65}
66//! Checks if a double is within eps of 0.0 [eps=1.0e-10 by default]
67inline constexpr bool zeroQ(double x, double eps = 1.0e-10) {
68 return x >= 0 ? x < eps : x > -eps;
69}

73//! @brief returns kappa_index, given kappa
74/*! @details
75 Kappa Index:
76 For easy array access, define 1-to-1 index for each kappa:
77
78 | kappa | -1 | 1 | -2 | 2 | -3 | 3 | -4 | 4 | ... |
79 |-------|----|---|----|---|----|---|----|---|-----|
80 | index | 0 | 1 | 2 | 3 | 4 | 5 | 6 | 7 | ... |
81
82 \f[
83 i_k(\kappa)=
84 \begin{cases}
85 -2\kappa-2, & \kappa<0, \\
86 2\kappa-1, & \kappa>0 .
87 \end{cases}
88 \f]
89
90 \f[
91 \kappa(i_k)=
92 \begin{cases}
93 -\left(\frac{i_k}{2}+1\right), & i_k\ \text{even},\\
94 \frac{i_k+1}{2}, & i_k\ \text{odd}.
95 \end{cases}
96 \f]
97*/
98constexpr std::uint64_t kappa_to_kindex(int ka) {
99 return (ka < 0) ? std::uint64_t(-2 * ka - 2) : std::uint64_t(2 * ka - 1);
100}
101
102//! Returns kappa, given kappa_index
103//! @details see \ref Angular::kappa_to_kindex()
104constexpr int kindex_to_kappa(std::uint64_t i) {
105 return (i % 2 == 0) ? int(-(i + 2) / 2) : int((i + 1) / 2);
106}
107
108//! returns 2j, given kappa_index
109//! @details see \ref Angular::kappa_to_kindex()
110constexpr int kindex_to_twoj(std::uint64_t i) {
111 return (i % 2 == 0) ? int(i + 1) : int(i);
112}
113
114//! returns l, given kappa_index
115//! @details see \ref Angular::kappa_to_kindex()
116constexpr int kindex_to_l(std::uint64_t i) {
117 return (i % 2 == 0) ? int(i / 2) : int((i + 1) / 2);
118}

120//==============================================================================
121//! Returns number of possible states _below_ given n (i.e., n' < n)
122/*! @details
123
124 This could be made more "compact" with a maximum l.
125 In practice, we go to large n, but never very large l.
126
127 @warning
128 - n must be >1 (no non-standard states)
129
130 Cannot overflow; converts to std::uint64_t.
131*/
132constexpr std::uint64_t states_below_n(int n) {
133 const std::uint64_t nn = static_cast<std::uint64_t>(n);
134 return nn * nn - 2 * nn + 1;
135}
136
137//! @brief Returns nk_index, given {n, kappa}
138/*! @details
139 For convenient array access we define a one-to-one mapping between the pair
140 \f$(n,\kappa)\f$ and a non-negative index:
141
142 \f[
143 i_{nk}(n,\kappa) = (n-1)^2 + i_k(\kappa),
144 \f]
145
146 where \f$i_k(\kappa)\f$ is the kappa index defined by \ref Angular::kappa_to_kindex()
147
148 Equivalently,
149
150 \f[
151 i_{nk}(n,\kappa) = n^2 - 2n + 1 + i_k(\kappa).
152 \f]
153
154 Since
155
156 \f[
157 n^2 - 2n + 1 = (n-1)^2 = \texttt{states\_below\_n}(n),
158 \f]
159
160 this counts the number of states with principal quantum number \f$n' < n\f$,
161 plus the index within the \f$n\f$ shell.
162
163 @note This indexing is only valid for physical bound-state quantum numbers, n >= 1,
164 and cannot in general be used for arbitrary basis states.
165
166 @warning To safely convert the returned index (`uint64_t`) to `int`, one must
167 have `n <= 46340`.
168
169 @warning To safely convert the returned index to \ref Coulomb::nkIndex
170 (`uint16_t`), one must have `n <= 256`.
171*/
172constexpr std::uint64_t nk_to_index(int n, int k) {
173 return states_below_n(n) + Angular::kappa_to_kindex(k);
174}
175
176//! Returns {n, kappa} given nk_index
177//! @details Inverse of \ref Angular::nk_to_index().
178inline std::pair<int, int> index_to_nk(std::uint64_t nk_index) {
179 // Better way? isqrt?
180 const auto n = 1 + int(std::sqrt(nk_index));
181 const auto kappa_index = nk_index - states_below_n(n);
182 return {n, Angular::kindex_to_kappa(kappa_index)};
183}
184
185//! Returns {n, kappa_index} given nk_index
186//! @details Inverse of \ref Angular::nk_to_index().
187inline std::pair<int, std::uint64_t>
188nkindex_to_n_kindex(std::uint64_t nk_index) {
189 const auto n = 1 + int(std::sqrt(nk_index));
190 const auto kappa_index = nk_index - states_below_n(n);
191 return {n, kappa_index};
192}
193
194//! Returns kappa, given nk_index
195//! @details See \ref Angular::nk_to_index().
196inline int nkindex_to_kappa(std::uint64_t nk_index) {
197 const auto n = 1 + int(std::sqrt(nk_index));
198 const auto kappa_index = nk_index - states_below_n(n);
199 return kindex_to_kappa(kappa_index);
200}
201
202//! Returns 2*j, given nk_index
203//! @details See \ref Angular::nk_to_index().
204inline int nkindex_to_twoj(std::uint64_t nk_index) {
205 const auto n = 1 + int(std::sqrt(nk_index));
206 const auto kappa_index = nk_index - states_below_n(n);
207 return kindex_to_twoj(kappa_index);
208}
209
210//! Returns l, given nk_index
211//! @details See \ref Angular::nk_to_index().
212inline int nkindex_to_l(std::uint64_t nk_index) {
213 const auto n = 1 + int(std::sqrt(double(nk_index) + 0.01));
214 const auto kappa_index = nk_index - states_below_n(n);
215 return l_k(kindex_to_kappa(kappa_index));
216}

219//! Returns true if a is even - for integer values
220constexpr bool evenQ(int a) { return (a % 2 == 0); }
221//! Returns true if a (an int) is even, given 2*a (true if two_a/2 is even)
222constexpr bool evenQ_2(int two_a) { return (two_a % 4 == 0); }
223//! Evaluates (-1)^{a} (for integer a)
224constexpr int neg1pow(int a) { return evenQ(a) ? 1 : -1; }
225//! Evaluates (-1)^{two_a/2} (for integer a; two_a is even)
226constexpr int neg1pow_2(int two_a) { return evenQ_2(two_a) ? 1 : -1; }

229//! Parity rule. Returns 1 only if la+lb+k is even
230constexpr int parity(int la, int lb, int k) {
231 return evenQ(la + lb + k) ? 1 : 0;
232}

235//! Returns 1 if triangle rule is satisfied. nb: works with j OR twoj!
236constexpr int triangle(int j1, int j2, int J) {
237 // nb: can be called with wither j or twoj!
238 return ((j1 + j2 < J) || (std::abs(j1 - j2) > J)) ? 0 : 1;
239}
240//! Checks if three ints sum to zero, returns 1 if they do
241constexpr int sumsToZero(int m1, int m2, int m3) {
242 return (m1 + m2 + m3 != 0) ? 0 : 1;
243}

246/*!
247 @brief Wigner 3j symbol. Inputs are 2*j (or 2*l) as integers.
248 @details
249 Computes the Wigner 3j symbol:
250 \f[
251 \begin{pmatrix} j_1 & j_2 & j_3 \\ m_1 & m_2 & m_3 \end{pmatrix}
252 \f]
253 Returns zero if the triangle rule or \f$ m_1+m_2+m_3=0 \f$ is not satisfied.
254 Wraps \c gsl_sf_coupling_3j.
255
256 @param two_j1 2*j1
257 @param two_j2 2*j2
258 @param two_j3 2*j3
259 @param two_m1 2*m1
260 @param two_m2 2*m2
261 @param two_m3 2*m3
262 @return Value of the 3j symbol; 0 if selection rules are not met.
263*/
264inline double threej_2(int two_j1, int two_j2, int two_j3, int two_m1,
265 int two_m2, int two_m3) {
266 if (triangle(two_j1, two_j2, two_j3) * sumsToZero(two_m1, two_m2, two_m3) ==
267 0)
268 return 0;
269 return gsl_sf_coupling_3j(two_j1, two_j2, two_j3, two_m1, two_m2, two_m3);
270}

273/*!
274 @brief Special 3j symbol: (j1 j2 k; -1/2 1/2 0). Inputs are 2*j as integers.
275 @details
276 Evaluates the commonly occurring 3j symbol:
277 \f[
278 \begin{pmatrix} j_1 & j_2 & k \\ -\tfrac{1}{2} & \tfrac{1}{2} & 0 \end{pmatrix}
279 \f]
280 Handles the \f$ k=0 \f$ case analytically; otherwise wraps \c gsl_sf_coupling_3j.
281 Returns zero if the triangle rule is not satisfied.
282
283 @param two_j1 2*j1
284 @param two_j2 2*j2
285 @param two_k 2*k
286 @return Value of the 3j symbol; 0 if selection rules are not met.
287*/
288inline double special_threej_2(int two_j1, int two_j2, int two_k) {
289 if (triangle(two_j1, two_j2, two_k) == 0)
290 return 0.0;
291 if (two_k == 0) {
292 auto s = evenQ_2(two_j1 + 1) ? 1.0 : -1.0;
293 return s / std::sqrt(two_j1 + 1);
294 }
295 return gsl_sf_coupling_3j(two_j1, two_j2, two_k, -1, 1, 0);
296}

299/*!
300 @brief Clebsch-Gordon coefficient <j1 m1, j2 m2 | J M>. Takes 2*j as integers.
301 @details
302 Computes the Clebsch-Gordon coefficient:
303 \f[
304 \langle j_1 m_1,\, j_2 m_2 \,|\, J M \rangle
305 = (-1)^{j_1-j_2+M}\sqrt{2J+1}
306 \begin{pmatrix} j_1 & j_2 & J \\ m_1 & m_2 & -M \end{pmatrix}
307 \f]
308 Returns zero if the triangle rule or \f$ m_1+m_2=M \f$ is not satisfied.
309
310 @param two_j1 2*j1
311 @param two_m1 2*m1
312 @param two_j2 2*j2
313 @param two_m2 2*m2
314 @param two_J 2*J
315 @param two_M 2*M
316 @return Value of the CG coefficient; 0 if selection rules are not met.
317*/
318inline double cg_2(int two_j1, int two_m1, int two_j2, int two_m2, int two_J,
319 int two_M) {
320 if (triangle(two_j1, two_j2, two_J) * sumsToZero(two_m1, two_m2, -two_M) == 0)
321 return 0;
322 int sign = -1;
323 if ((two_j1 - two_j2 + two_M) % 4 == 0)
324 sign = 1; // mod 4 (instead 2), since x2
325 return sign * std::sqrt(two_J + 1.) *
326 gsl_sf_coupling_3j(two_j1, two_j2, two_J, two_m1, two_m2, -two_M);
327}

330/*!
331 @brief Returns true if the 6j symbol is zero by triangle/parity rules. Inputs are 2*j.
332 @details
333 Checks all four triangle triads and integer-sum (parity) conditions for:
334 \f[
335 \sixj{a}{b}{c}{d}{e}{f}
336 \f]
337 The triads (abc), (cde), (bdf), (efa) must each satisfy the triangle rule
338 and sum to an even integer (inputs are 2*j).
339
340 @note Some symbols that pass these checks are still numerically zero.
341 GSL may return a very small non-zero value in such cases.
342 @warning Only valid for inputs that are 2*(integer or half-integer).
343*/
344inline bool sixj_zeroQ(int a, int b, int c, int d, int e, int f) {
345 if (triangle(a, b, c) == 0)
346 return true;
347 if (triangle(c, d, e) == 0)
348 return true;
349 if (triangle(b, d, f) == 0)
350 return true;
351 if (triangle(e, f, a) == 0)
352 return true;
353 if (!evenQ(a + b + c))
354 return true;
355 if (!evenQ(c + d + e))
356 return true;
357 if (!evenQ(b + d + f))
358 return true;
359 if (!evenQ(e + f + a))
360 return true;
361 return false;
362}

364/*!
365 @brief Checks 6j triangle conditions with optional arguments (wildcards). Inputs are 2*j.
366 @details
367 Checks only the triads formed by the provided (non-null) arguments.
368 Missing (empty) optionals act as wildcards.
369 For example, \c sixjTriads(a,b,c,{},{},{}) checks whether any 6j symbol
370 \f$ \{ a\, b\, c \;|\; *\, *\, * \} \f$ could be non-zero.
371
372 The four triads of \f$ \{a\,b\,c\;|\;d\,e\,f\} \f$ are: (abc), (cde), (aef), (bdf).
373
374 @param a 2*j1 (optional)
375 @param b 2*j2 (optional)
376 @param c 2*j3 (optional)
377 @param d 2*j4 (optional)
378 @param e 2*j5 (optional)
379 @param f 2*j6 (optional)
380 @return false if any complete triad fails the triangle rule; true otherwise.
381*/
382inline bool sixjTriads(std::optional<int> a, std::optional<int> b,
383 std::optional<int> c, std::optional<int> d,
384 std::optional<int> e, std::optional<int> f) {
385 // nb: works for 2*integers ONLY
386 // checks triangle consitions, with optional parameters
387 if (a && b && c && triangle(*a, *b, *c) == 0)
388 return false;
389 if (c && d && e && triangle(*c, *d, *e) == 0)
390 return false;
391 if (a && e && f && triangle(*a, *e, *f) == 0)
392 return false;
393 if (b && d && f && triangle(*b, *d, *f) == 0)
394 return false;
395
396 return true;
397}

454/*!
455 @brief Reduced relativistic angular ME: <ka||C^k||kb>. Takes kappa values.
456 @details
457 Computes the reduced matrix element of the spherical tensor \f$ C^k \f$
458 between relativistic states labelled by Dirac quantum numbers:
459 \f[
460 \langle \kappa_a \| C^k \| \kappa_b \rangle
461 = (-1)^{j_a+\tfrac{1}{2}} \sqrt{[j_a][j_b]}
462 \begin{pmatrix} j_a & j_b & k \\ -\tfrac{1}{2} & \tfrac{1}{2} & 0 \end{pmatrix}
463 \pi(l_a+l_b+k),
464 \f]
465 where \f$ [j] \equiv 2j+1 \f$ and \f$ \pi(l_a+l_b+k) \f$ is 1 if \f$ l_a+l_b+k \f$
466 is even, 0 otherwise.
467
468 @param k Multipole rank.
469 @param ka Dirac quantum number \f$ \kappa_a \f$ (bra).
470 @param kb Dirac quantum number \f$ \kappa_b \f$ (ket).
471 @return \f$ \langle \kappa_a \| C^k \| \kappa_b \rangle \f$; 0 if selection rules are not met.
472*/
473inline double Ck_kk(int k, int ka, int kb) {
474 if (parity(l_k(ka), l_k(kb), k) == 0) {
475 return 0.0;
476 }
477 auto two_ja = twoj_k(ka);
478 auto two_jb = twoj_k(kb);
479 // auto sign = ((two_ja + 1) / 2 % 2 == 0) ? 1 : -1;
480 auto sign = evenQ_2(two_ja + 1) ? 1 : -1;
481 auto f = std::sqrt((two_ja + 1) * (two_jb + 1));
482 auto g = special_threej_2(two_ja, two_jb, 2 * k);
483 return sign * f * g;
484}

486/*!
487 @brief Full (non-reduced) C^k_q matrix element. Takes kappa and 2*m values.
488 @details
489 Computes the full matrix element including magnetic quantum numbers via
490 the Wigner-Eckart theorem:
491 \f[
492 \langle \kappa_a m_a | C^k_q | \kappa_b m_b \rangle
493 = (-1)^{j_a-m_a}
494 \begin{pmatrix} j_a & k & j_b \\ -m_a & q & m_b \end{pmatrix}
495 \langle \kappa_a \| C^k \| \kappa_b \rangle
496 \f]
497
498 @param k Multipole rank.
499 @param ka Dirac quantum number \f$ \kappa_a \f$ (bra).
500 @param kb Dirac quantum number \f$ \kappa_b \f$ (ket).
501 @param twoma 2*m_a
502 @param twomb 2*m_b
503 @param twoq 2*q
504 @return \f$ \langle \kappa_a m_a | C^k_q | \kappa_b m_b \rangle \f$
505*/
506inline double Ck_kk_mmq(int k, int ka, int kb, int twoma, int twomb, int twoq) {
507 const auto tja = twoj_k(ka);
508 const auto tjb = twoj_k(kb);
509 return neg1pow_2(tja - twoma) *
510 threej_2(tja, 2 * k, tjb, -twoma, twoq, twomb) * Ck_kk(k, ka, kb);
511}

513/*!
514 @brief Selection rule check for C^k: returns true if <ka||C^k||kb> may be non-zero.
515 @details
516 Checks parity and triangle conditions only; does not compute the value.
517 Equivalent to \c Ck_kk(k,ka,kb)!=0 but avoids the full computation.
518
519 @param k Multipole rank.
520 @param ka Dirac quantum number \f$ \kappa_a \f$.
521 @param kb Dirac quantum number \f$ \kappa_b \f$.
522 @return false if \f$ \langle \kappa_a \| C^k \| \kappa_b \rangle = 0 \f$ by selection rules; true otherwise.
523*/
524inline bool Ck_kk_SR(int k, int ka, int kb) {
525 if (parity(l_k(ka), l_k(kb), k) == 0)
526 return false;
527 if (triangle(twoj_k(ka), twoj_k(kb), 2 * k) == 0)
528 return false;
529 return true;
530}

532/*!
533 @brief Tilde variant of the C^k reduced ME: (-1)^{ja+1/2} * <ka||C^k||kb>
534 @details
535 Computes \f$ \widetilde{C}^k_{ab} \equiv (-1)^{j_a+1/2} \langle \kappa_a \| C^k \| \kappa_b \rangle \f$.
536 This removes the \f$ (-1)^{j_a+1/2} \f$ phase from \ref Ck_kk, leaving
537 the purely geometric 3j piece times \f$ \sqrt{[j_a][j_b]} \f$.
538
539 @param k Multipole rank.
540 @param ka Dirac quantum number \f$ \kappa_a \f$.
541 @param kb Dirac quantum number \f$ \kappa_b \f$.
542 @return \f$ \widetilde{C}^k_{ab} \f$
543*/
544inline double tildeCk_kk(int k, int ka, int kb) {
545 auto m1tjph = evenQ_2(twoj_k(ka) + 1) ? 1 : -1;
546 return m1tjph * Ck_kk(k, ka, kb);
547}

549/*!
550 @brief Min and max non-zero multipole rank k for C^k, given kappa_a and kappa_b.
551 @details
552 Returns \f$ (k_\text{min}, k_\text{max}) \f$ such that
553 \f$ \langle \kappa_a \| C^k \| \kappa_b \rangle \neq 0 \f$ only for
554 \f$ k_\text{min} \leq k \leq k_\text{max} \f$.
555 Steps are in increments of 2, enforced by parity.
556
557 Derived from:
558 \f[
559 k_\text{min} = |j_a - j_b|, \quad k_\text{max} = j_a + j_b,
560 \f]
561 then adjusted so that \f$ l_a + l_b + k \f$ is even.
562
563 @param ka Dirac quantum number \f$ \kappa_a \f$.
564 @param kb Dirac quantum number \f$ \kappa_b \f$.
565 @return \f$ \{k_\text{min},\, k_\text{max}\} \f$
566*/
567inline std::pair<int, int> kminmax_Ck(int ka, int kb) {
568 // j = |k|-0.5
569 // kmin = |ja-jb| = | |ka| - |kb| |
570 // kmax = ja+jb = |ka| + |kb| - 1
571 const auto aka = std::abs(ka);
572 const auto akb = std::abs(kb);
573
574 auto kmin = std::abs(aka - akb);
575 auto kmax = aka + akb - 1;
576 // account for parity
577 if (parity(l_k(ka), l_k(kb), kmin) == 0)
578 ++kmin;
579 if (parity(l_k(ka), l_k(kb), kmax) == 0)
580 --kmax;
581
582 return {kmin, kmax};
583}

585//==============================================================================
586/*!
587 @brief Reduced spin-1/2 angular ME: <ka||S||kb>. Takes kappa values.
588 @details
589 Computes the reduced matrix element of the spin operator \f$ \mathbf{S} \f$
590 (for spin-1/2) between relativistic states:
591 \f[
592 \langle \kappa_a \| S \| \kappa_b \rangle
593 = \delta_{l_a l_b} \, (-1)^{j_a+l_a+3/2} \sqrt{[j_a][j_b]\tfrac{3}{2}}
594 \sixj{j_a}{1}{j_b}{\tfrac{1}{2}}{l_a}{\tfrac{1}{2}}
595 \f]
596 The 6j symbol is evaluated analytically:
597 \f[
598 \sixj{j_a}{1}{j_b}{\tfrac{1}{2}}{l_a}{\tfrac{1}{2}}
599 = \frac{(-1)^{\kappa_a+\kappa_b}}{2}
600 \sqrt{\left|\frac{(\kappa_a-1)^2 - \kappa_b^2}{3\kappa_a(1+2\kappa_a)}\right|}
601 \f]
602 This special-case formula is ~20% faster than the general 6j routine.
603 Returns 0 if \f$ l_a \neq l_b \f$ or the triangle rule is violated.
604
605 @param ka Dirac quantum number \f$ \kappa_a \f$.
606 @param kb Dirac quantum number \f$ \kappa_b \f$.
607 @return \f$ \langle \kappa_a \| S \| \kappa_b \rangle \f$; 0 if selection rules are not met.
608*/
609inline double S_kk(int ka, int kb) {
610 auto la = l_k(ka);
611 if (la != l_k(kb))
612 return 0.0;
613 auto tja = twoj_k(ka);
614 auto tjb = twoj_k(kb);
615 if (triangle(tja, 2, tjb) == 0)
616 return 0;
617 auto sign = (((tja + 2 * la + 3) / 2) % 2 == 0) ? 1 : -1;
618 auto f = std::sqrt((tja + 1) * (tjb + 1) * 1.5);
619 // auto sixj = gsl_sf_coupling_6j(tja, 2, tjb, 1, 2 * la, 1);
620 auto sixj_sign = ((ka + kb) % 2 == 0) ? 1.0 : -1.0;
621 auto sixj = 0.5 * sixj_sign *
622 std::sqrt(std::fabs(((ka - 1) * (ka - 1) - kb * kb) /
623 (3.0 * ka * (1.0 + 2.0 * ka))));
624 return sign * f * sixj;
625}
