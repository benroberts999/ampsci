79 DiracSpinor act(const DiracSpinor &Fa) const {
80 using namespace qip::overloads;
81 auto mFa = Fa;
82 // mFa.f() = ff * Fa.f() + fg * Fa.g();
83 // mFa.g() = gf * Fa.f() + gg * Fa.g();
84
85 // since we start with a copy:
86 if (ff != 1.0)
87 mFa.f() *= ff;
88 if (fg != 0.0)
89 mFa.f() += fg * Fa.g();
90 if (gg != 1.0)
91 mFa.g() *= gg;
92 if (gf != 0.0)
93 mFa.g() += gf * Fa.f();
94 return mFa;
95 }
