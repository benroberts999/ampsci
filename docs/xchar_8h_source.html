34 inline auto write_loc(std::back_insert_iterator<detail::buffer<wchar_t>> out,
35  loc_value value, const format_specs<wchar_t> &specs,
36  locale_ref loc) -> bool {
37 #ifndef FMT_STATIC_THOUSANDS_SEPARATOR
38  auto &numpunct =
39  std::use_facet<std::numpunct<wchar_t>>(loc.get<std::locale>());
40  auto separator = std::wstring();
41  auto grouping = numpunct.grouping();
42  if (!grouping.empty())
43  separator = std::wstring(1, numpunct.thousands_sep());
44  return value.visit(loc_writer<wchar_t>{out, specs, separator, grouping, {}});
45 #endif
46  return false;
47 }
