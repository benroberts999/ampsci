153template <class A>
154static int twojk(const A &a) {
155 if constexpr (std::is_same_v<A, DiracSpinor>) {
156 return a.twoj();
157 } else {
158 static_assert(std::is_same_v<A, int>);
159 return 2 * a;
160 }
161}
162
163template <class A>
164static std::optional<int> twojknull(const A &a) {
165 if constexpr (std::is_same_v<A, DiracSpinor>) {
166 return a.twoj();
167 } else if constexpr (std::is_same_v<A, int>) {
168 static_assert(std::is_same_v<A, int>);
169 return 2 * a;
170 } else {
171 return std::nullopt;
172 }
173}
